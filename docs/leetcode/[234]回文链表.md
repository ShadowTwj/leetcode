    public boolean isPalindrome(ListNode head) {
            List<Integer> data = new ArrayList<>();
            while (head != null) {
                data.add(head.val);
                head = head.next;
            }
            boolean flag = true;
            for (int i = 0; i < data.size() / 2; i++) {
                if (!data.get(i).equals(data.get(data.size() - i - 1))) {
                    flag = false;
                }
            }
            return flag;
        }
    }
