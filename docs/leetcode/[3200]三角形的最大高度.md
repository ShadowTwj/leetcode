class Solution {
    public int maxHeightOfTriangle(int red, int blue) {
        return Math.max(height(red, blue), height(blue, red));
    }
    public int height(int first, int second) {
        int firstStart = 1;
        int secondStart = 2;
        int res = 0;
        do {
            if (firstStart <= first) {
                res++;
                first -= firstStart;
                firstStart += 2;
            } else {
                return res;
            }
            if (secondStart <= second) {
                res++;
                second -= secondStart;
                secondStart += 2;
            } else {
                return res;
            }
        } while (true);
    }
}
