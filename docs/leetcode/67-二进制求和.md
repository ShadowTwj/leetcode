class Solution {
    public String addBinary(String a, String b) {
        StringBuilder sb = new StringBuilder();
        int sum = 0;
        for (int i = a.length() - 1, j = b.length() - 1; i >= 0 || j >= 0; i--, j--) {
            sum += i < 0 || a.charAt(i) == '0' ? 0 : 1;
            sum += j < 0 || b.charAt(j) == '0' ? 0 : 1;

            sb.append(sum % 2);
            sum = sum / 2;
        }

        if (sum > 0) {
            sb.append(sum % 2);
        }

        return sb.reverse().toString();
    }
}
