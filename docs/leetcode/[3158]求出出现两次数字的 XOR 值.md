class Solution {
    public int duplicateNumbersXOR(int[] nums) {
        int result = 0;
        Set<Integer> set = new HashSet<>();
        for (int num : nums) {
            if (set.contains(num)) {
                result ^= num;
                set.remove(num);
            } else {
                set.add(num);
            }
        }

        return result;
    }
}
